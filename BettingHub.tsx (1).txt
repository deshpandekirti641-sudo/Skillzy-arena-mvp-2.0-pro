const BET_AMOUNT = 10; // Only ‚Çπ10 betting for all games

  playerProfile, 
  playerWallet, 
  developerEarnings,
  onFindMatch, 
  onDeposit 
}: BettingHubProps) {
  const [selectedGame, setSelectedGame] = useState<string | null>(null);
  const selectedBet = BET_AMOUNT; // Fixed ‚Çπ10 bet

  const balance = Number(playerWallet.balanceCents) / 100;
  const winRate = playerProfile.totalMatches > 0 ? 
    (playerProfile.wins / playerProfile.totalMatches * 100).toFixed(1) : '0.0';
  
  // Format balance in rupees
  const formatBalance = (cents: bigint) => `‚Çπ${(Number(cents) / 100).toFixed(2)}`;
  const balanceRupees = formatBalance(playerWallet.balanceCents);

  const handleFindMatch = () => {
    if (!selectedGame) return;
    
    const gameType = { tag: selectedGame } as GameType;
    onFindMatch(gameType, selectedBet);
  };

  const canAffordBet = balance >= selectedBet;

  return (
    <div className="min-h-screen bg-gradient-to-br from-purple-900 via-blue-900 to-indigo-900 p-4">
      {/* Header */}
      <div className="max-w-6xl mx-auto mb-8">
        <div className="text-center mb-6">
          <h1 className="text-4xl font-bold text-white mb-2">üí∞ Skillzy Arena</h1>
          <p className="text-xl text-white/80">Real Money 1v1 Skill Battles</p>
        </div>

        {/* Player Stats */}
        <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-8">
          <Card className="bg-white/10 backdrop-blur-md border-white/20 text-white">
            <CardContent className="p-4 text-center">
              <p className="text-sm opacity-80">Balance</p>
              <p className="text-2xl font-bold text-green-400">{balanceRupees}</p>
            </CardContent>
          </Card>
          
          <Card className="bg-white/10 backdrop-blur-md border-white/20 text-white">
            <CardContent className="p-4 text-center">
              <p className="text-sm opacity-80">Matches</p>
              <p className="text-2xl font-bold">{playerProfile.totalMatches}</p>
            </CardContent>
          </Card>
          
          <Card className="bg-white/10 backdrop-blur-md border-white/20 text-white">
            <CardContent className="p-4 text-center">
              <p className="text-sm opacity-80">Win Rate</p>
              <p className="text-2xl font-bold text-yellow-400">{winRate}%</p>
            </CardContent>
          </Card>
          
          <Card className="bg-white/10 backdrop-blur-md border-white/20 text-white">
            <CardContent className="p-4 text-center">
              <p className="text-sm opacity-80">Record</p>
              <p className="text-lg font-bold">
                <span className="text-green-400">{playerProfile.wins}W</span>
                <span className="text-white/60"> - </span>
                <span className="text-red-400">{playerProfile.losses}L</span>
              </p>
            </CardContent>
          </Card>
        </div>
      </div>

      {/* Game Selection */}
      <div className="max-w-4xl mx-auto">
        <Card className="bg-white/10 backdrop-blur-md border-white/20 text-white">
          <CardHeader>
            <CardTitle className="text-center text-2xl">Choose Your Battle</CardTitle>
            <p className="text-center text-white/80">Select game and bet amount to find an opponent</p>
          </CardHeader>
          <CardContent className="space-y-6">
            
            {/* Games Grid */}
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              {GAMES.map((game) => (
                <Card
                  key={game.id}
                  className={`cursor-pointer transition-all transform hover:scale-105 ${
                    selectedGame === game.id 
                      ? 'ring-4 ring-yellow-400 bg-white/20' 
                      : 'bg-white/5 hover:bg-white/10'
                  } border-white/20`}
                  onClick={() => setSelectedGame(game.id)}
                >
                  <CardContent className="p-6 text-center">
                    <div className="text-4xl mb-2">{game.icon}</div>
                    <h3 className="text-xl font-bold text-white mb-2">{game.title}</h3>
                    <p className="text-white/70">{game.description}</p>
                  </CardContent>
                </Card>
              ))}
            </div>

            {/* Fixed Bet Amount Display */}
            {selectedGame && (
              <div className="space-y-4">
                <h3 className="text-xl font-semibold text-center">üéØ Universal Bet Amount</h3>
                <div className="flex justify-center">
                  <div className="bg-gradient-to-r from-yellow-500 to-orange-500 p-6 rounded-xl text-center max-w-xs">
                    <div className="text-3xl font-bold text-black mb-2">‚Çπ{BET_AMOUNT}</div>
                    <div className="text-black/80 font-medium mb-2">Fixed bet for all games</div>
                    <div className="text-sm text-black/70">
                      Winner gets: <span className="font-bold">‚Çπ{(BET_AMOUNT * 1.6).toFixed(0)}</span>
                    </div>
                  </div>
                </div>
                <p className="text-center text-sm text-white/70">
                  üí∞ Same bet, same excitement ‚Ä¢ Winner gets 80% of total pot
                </p>
              </div>
            )}

            {/* Action Buttons */}
            <div className="flex justify-center space-x-4 pt-4">
              <Button
                onClick={handleFindMatch}
                disabled={!selectedGame || !canAffordBet}
                className="px-8 py-4 text-lg font-bold bg-gradient-to-r from-green-500 to-blue-500 hover:from-green-600 hover:to-blue-600 text-white disabled:opacity-50"
              >
                üéØ Find Match (‚Çπ{BET_AMOUNT})
              </Button>
              
              <Button
                onClick={() => onDeposit(1000)} 
                variant="outline"
                className="px-6 py-4 text-lg font-semibold bg-white/10 text-white hover:bg-white/20 border-white/30"
              >
                üí≥ Add ‚Çπ10.00
              </Button>
            </div>

            {/* Insufficient Balance Warning */}
            {!canAffordBet && selectedGame && (
              <div className="text-center p-4 bg-red-500/20 border border-red-500/30 rounded-lg">
                <p className="text-red-200">
                  ‚ö†Ô∏è Insufficient balance! You need ‚Çπ{BET_AMOUNT} but only have {balanceRupees}
                </p>
                <p className="text-red-200/80 text-sm mt-1">
                  Add money to your wallet to start playing!
                </p>
              </div>
            )}

            {/* Platform Stats */}
            {developerEarnings && (
              <div className="text-center pt-6 border-t border-white/20">
                <p className="text-white/60 text-sm">
                  Platform Revenue: {formatBalance(developerEarnings.totalFeesCollectedCents)}
                </p>
              </div>
            )}
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
